# Type here!

  root:   Math.sqrt

 coffeescript-to-javascript.js 
(function() {
  var math;

  math = {
    root: Math.sqrt,
    square: square,
    cube: function(x) {
      return x * square(x);
    }
  };

  alert("Three cubed is " + (math.cube(3)));

}).call(this);

Roy to JavaScript
roy-to-javascript.js.roy 
let gcd a b =
  if b == 0 then
    a
  else
    gcd b (a % b)

console.log (gcd 49 35)
 roy-to-javascript.js 
(function() {
var gcd = function(a, b) {
    return (function() {
        if(b == 0) {
            return a;
        } else {
            return gcd(b, (a % b));
        }
    })();
};
console.log((gcd(49, 35)));
})();
